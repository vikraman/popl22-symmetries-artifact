quote₀ : UFin[ n ] → U n
quote₀ = quote^₀ ∘ quoteNorm₀

eval₀ : U n → UFin[ n ]
eval₀ = evalNorm₀ ∘ eval^₀

quote-eval₀ : (t : U n) → quote₀ (eval₀ t) ⟷₁ t
quote-eval₀ t = quote^₁ (quote-evalNorm₀ (eval^₀ t)) ◎ quote-eval^₀ t

eval-quote₀ : {n : ℕ} (X : UFin[ n ]) → Trunc -1 (eval₀ (quote₀ X) == X)
eval-quote₀ = eval-quoteNorm₀
